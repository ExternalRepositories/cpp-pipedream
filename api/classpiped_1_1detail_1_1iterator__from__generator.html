# piped::detail::iterator_from_generator

# <piped/generator.hpp>

template<typename T>
class piped::detail::iterator_from_generator;

generator<T> builds a C++-style "container" using this, consisting of the appropriate methods and a hack to test for end. You should generally not have to use this explicitly.
